func bubble(int a[],int s)
{
 var i,j;
 var temp int;
 i = 1;
 while(i<s)
  {
      j = 0;
    while(j<s-i)
     {
      if(a[j]>a[j+1])
       {
        temp=a[j];
        a[j]=a[j+1];
        a[j+1]=temp;
       }
       j++;
     }
     i++;
   }
}
